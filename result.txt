cd ~/Desktop/projects/fileshare
wg genkey | tee android_client_private.key | wg pubkey | tee android_client_public.key
# Add client to WireGuard
sudo wg set wg0 peer $(cat android_client_public.key) allowed-ips 10.0.0.2/32

# Save configuration
sudo wg-quick save wg0

# Verify it was added
sudo wg show wg0

cat > android_client.conf <<EOF
[Interface]
PrivateKey = $(cat android_client_private.key)
Address = 10.0.0.2/24
DNS = 8.8.8.8

[Peer]
PublicKey = c+yzw8EMaxlzAPm48NmYKBZkEErYhlX6d2Rs1D0jRRs=
Endpoint = 102.219.208.122:51820
AllowedIPs = 0.0.0.0/0
PersistentKeepalive = 25
EOF
